#include <cstdio>
#include <cstring>
using namespace std;

int N, len;
char number[10010];

double solve(int L, int R) {
    double sum = 0;
    for(int i=L; i<=R; i++) {
        sum = sum * 10 + (number[i] - '0');  //ÇØ¾ÅÉØ
    }
    //printf(" >>%lf<< ",sum);
    return sum;
}

int main() {
    scanf("%d",&N);
    for(int i=0; i<N; i++) {
        scanf(" %s",&number);
        len = strlen(number);
        double x = solve(0,len-1) / (solve(0,len/2-1) * solve(len/2,len-1));
        if(x == (int)x) {
            printf("Yes\n");
        } else {
            printf("No\n");
        }
    }

    return 0;
}
